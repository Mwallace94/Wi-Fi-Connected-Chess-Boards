void move_home() {
    Slp_1_Write(1);
    Slp_2_Write(1);
    
    uint8 x = Lim_1_Read(), y = Lim_2_Read();
    Debug_UART_PutString("limit start");
    
    while(!x || !y) {    
        if(!y){
            Dir_1_Write(1);
            Dir_2_Write(1);
            
            Stp_1_Write(0);
            Stp_1_Write(1);
            Stp_2_Write(0);
            Stp_2_Write(1);
            CyDelay(10);
        }      
        else if(!x){
            Dir_1_Write(0);
            Dir_2_Write(1);
            
            Stp_1_Write(0);
            Stp_1_Write(1);
            Stp_2_Write(0);
            Stp_2_Write(1);
            CyDelay(10);
        }
        x = Lim_1_Read();
        y = Lim_2_Read();
        
        if(x){
         Debug_UART_PutString("limit 1 read");   
        }
        if(y){
         Debug_UART_PutString("limit 2 read");   
        }
    }
    Slp_1_Write(0);
    Slp_2_Write(0);
    
    x_pos = 0;
    y_pos = 0;
}

void move_x(int16 mm) { 
    
    if(x_pos + mm > MAX_X_POS || x_pos + mm < MIN_X_POS) {
        return;
    }
    
    x_pos += mm;
    
    Slp_1_Write(1);
    Slp_2_Write(1);
    if(mm < 0) {
        Dir_1_Write(1);
        Dir_2_Write(1);
        mm *= -1;
    } else {
        Dir_1_Write(0);
        Dir_2_Write(0);
    }
    int j;
    
    for(j = 0; j < STEPS_PER_MM * mm * .5; j++) {
        Stp_1_Write(0);
        Stp_1_Write(1);
        Stp_2_Write(0);
        Stp_2_Write(1);
        CyDelay(10);
    }
    
    Slp_1_Write(0);
    Slp_2_Write(0);
}
    
void move_y(int16 mm) { 
        
    if(y_pos + mm > MAX_Y_POS || y_pos + mm < MIN_Y_POS) {
        return;
    }
    
    Slp_1_Write(1);
    Slp_2_Write(1);
    
    y_pos += mm;
    
    if(mm < 0) {
        Dir_1_Write(0);
        Dir_2_Write(1);
        mm *= -1;
    } else {
        Dir_1_Write(1);
        Dir_2_Write(0);
    }
    int j;
    
    for(j = 0; j < STEPS_PER_MM * mm * .5; j++) {
        Stp_1_Write(0);
        Stp_1_Write(1);
        Stp_2_Write(0);
        Stp_2_Write(1);
        CyDelay(20);
    }
    
    Slp_1_Write(0);
    Slp_2_Write(0);
}
